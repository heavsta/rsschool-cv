```go
// Bubble Sorting Algorithm in Go

func BubbleSort(sli []int) {
	l := len(sli)

	for i := range sli[:l] {
		for j := range sli[:l-i-1] {
			if sli[j] > sli[j+1] {
				Swap(sli, j)
			}
		}
	}
}

func Swap(sli []int, i int) {
	sli[i], sli[i+1] = sli[i+1], sli[i]
}
```
